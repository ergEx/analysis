# Fit condition specific params for isoelastic model for each subject using no pooling of information

model{

##LIKELIHOOD
for (c in 1:nConditions){
for (i in 1:nSubjects){
oneminEta[i,c] = 1 - eta_i[i,c]
for (t in 1:nTrials){
    wLU[i,c,t]    = w[i,c,t] + dwLU[i,c,t]
    wLL[i,c,t]    = w[i,c,t] + dwLL[i,c,t]
    wRU[i,c,t]    = w[i,c,t] + dwRU[i,c,t]
    wRL[i,c,t]    = w[i,c,t] + dwRL[i,c,t]
    wLUp[i,c,t]   = ifelse(wLU[i,c,t] > 0, wLU[i,c,t], 1)
    wLLp[i,c,t]   = ifelse(wLL[i,c,t] > 0, wLL[i,c,t], 1)
    wRUp[i,c,t]   = ifelse(wRU[i,c,t] > 0, wRU[i,c,t], 1)
    wRLp[i,c,t]   = ifelse(wRL[i,c,t] > 0, wRL[i,c,t], 1)
    wp[i,c,t]     = ifelse(w[i,c,t] > 0, w[i,c,t], 1)
    u[i,c,t]      = (pow(wp[i,c,t],oneminEta[i,c]) - 1) / oneminEta[i,c]
    uLU[i,c,t]    = (pow(wLUp[i,c,t],oneminEta[i,c]) - 1) / oneminEta[i,c]
    uLL[i,c,t]    = (pow(wLLp[i,c,t],oneminEta[i,c]) - 1) / oneminEta[i,c]
    uRU[i,c,t]    = (pow(wRUp[i,c,t],oneminEta[i,c]) - 1) / oneminEta[i,c]
    uRL[i,c,t]    = (pow(wRLp[i,c,t],oneminEta[i,c]) - 1) / oneminEta[i,c]
    duLU[i,c,t]   = uLU[i,c,t] - u[i,c,t]
    duLL[i,c,t]   = uLL[i,c,t] - u[i,c,t]
    duRU[i,c,t]   = uRU[i,c,t] - u[i,c,t]
    duRL[i,c,t]   = uRL[i,c,t] - u[i,c,t]
    eduL[i,c,t]   = (duLU[i,c,t] + duLL[i,c,t]) / 2
    eduR[i,c,t]   = (duRU[i,c,t] + duRL[i,c,t]) / 2
    deu[i,c,t]    = eduL[i,c,t] - eduR[i,c,t]
    sdeu[i,c,t]   = -1 * beta_i[i,c] * deu[i,c,t]
    cp[i,c,t]     = 1 / (1 + (exp(sdeu[i,c,t])))
    theta[i,c,t]  = max( 0.000001 , min(0.999999 , cp[i,c,t]) )
    y[i,c,t]      ~ dbern(theta[i,c,t])
}# end of trials
}# end of conditions
}# end of subjects


## PRIORS
for (c in 1:nConditions){
eta_g[c]       ~ dnorm(0,1)   # Not used
log_beta_g[c]  ~ dnorm(0,1)   # Not used
beta_g[c]      = exp(1)       # Not used

for (i in 1:nSubjects){
    log_beta_i[i,c]  ~ dnorm(mu_log_beta[i,c], tau_log_beta[i,c])
    beta_i[i,c]      = exp(log_beta_i[i,c])
    eta_i[i,c]       ~ dnorm(mu_eta[i,c],tau_eta[i,c])
}#end of subjects
}#end of conditions


##HYPERPRIORS
for (c in 1:nConditions){
for (i in 1:nSubjects){
    #beta parameter
    mu_log_beta[i,c]       ~ dunif(mu_log_beta_l,mu_log_beta_h)
    tau_log_beta[i,c]      = pow(sigma_log_beta[i,c],-2)
    sigma_log_beta[i,c]    ~ dunif(sigma_l,sigma_h)

    #eta parameter
    mu_eta[i,c]            ~ dunif(mu_eta_l,mu_eta_h)
    tau_eta[i,c]           = pow(sigma_eta[i,c],-2)
    sigma_eta[i,c]         ~ dunif(sigma_l,sigma_h)
}#end of subjects
}#end of conditions

}
